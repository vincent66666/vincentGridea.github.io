使用CODING 自动构建部署 Hyperf项目

作者: 我的测试学习 日期: 2022-03-29 阅读时间:8.1 分钟 字数:1554 分类: Devops Jenkins Hyperf 阅读量: -

不是集群部署，适合个人或小公司项目使用

创建基础项目

1、先创建一个项目

2、关联代码仓库

构建计划

整体的Jenkins流程

  1. 检出分支
  2. 载入配置信息
  3. 根据分支下的 Dockerfile 生成镜像
  4. 推送镜像到 coding 的制品仓库
  5. 最后登录服务器拉取镜像进行部署
    那么我们就开始编写构建计划吧

编写构建计划

创建构建计划

首先先创建一个构建计划

我们选择自定义构建过程

选择关联的代码仓库，配置来源选择使用静态配置的 Jenkinsfile

然后点击确认进来配置详情选择文本编辑器

载入配置信息并生成镜像并推送到制品仓库

pipeline {
  agent any
  stages {
    stage('检出') {
      steps {
        checkout([
          $class: 'GitSCM',
          branches: [[name: GIT_BUILD_REF]],
          userRemoteConfigs: [[
            url: GIT_REPO_URL,
            credentialsId: CREDENTIALS_ID
          ]]])
        }
      }

    stage('载入配置信息') {
      steps {
        sh "cp ${HYPERF_ENV_NAME} .env"
      }
    }
    stage('构建镜像并推送到 CODING Docker 制品库') {
      steps {
        sh "docker build -t ${CODING_DOCKER_IMAGE_NAME}:${DOCKER_IMAGE_VERSION} -f ${DOCKERFILE_PATH} ${DOCKER_BUILD_CONTEXT}"
        useCustomStepPlugin(key: 'coding-public:artifact_docker_push', version: 'latest', params: [image:"${CODING_DOCKER_IMAGE_NAME}:${DOCKER_IMAGE_VERSION}",repo:'hyperf',properties:'[]',project:'xuexi',username:'${PROJECT_TOKEN_GK}',password:'${PROJECT_TOKEN}'])
      }
    }
  }
}

自动部署

pipeline {
  agent any
  stages {
    stage('检出') {
      steps {
        checkout([$class: 'GitSCM',
        branches: [[name: GIT_BUILD_REF]],
        userRemoteConfigs: [[
          url: GIT_REPO_URL,
          credentialsId: CREDENTIALS_ID
        ]]])
      }
    }
    stage('代码检测') {
      steps {
        sh 'php -v'
      }
    }
    stage('载入配置信息') {
      steps {
        sh "cp ${HYPERF_ENV_NAME} .env"
      }
    }
    stage('构建镜像并推送到 CODING Docker 制品库') {
      steps {
        sh 'pwd && ls -l'
        useCustomStepPlugin(key: 'coding-public:magic-version', version: 'latest')
        script {
          readProperties(file: env.CI_ENV_FILE).each {key, value -> env[key] = value }
        }
        echo "${env.MAGIC_VERSION}"
        sh "docker build -t ${CODING_DOCKER_IMAGE_NAME}:${DOCKER_IMAGE_VERSION} -f ${DOCKERFILE_PATH} ${DOCKER_BUILD_CONTEXT}"
        useCustomStepPlugin(key: 'coding-public:artifact_docker_push', version: 'latest', params: [image:"${CODING_DOCKER_IMAGE_NAME}:${DOCKER_IMAGE_VERSION}",repo:'hyperf',properties:'[]',project:'xuexi',username:'${PROJECT_TOKEN_GK}',password:'${PROJECT_TOKEN}'])
      }
    }
    stage('部署到远端服务') {
      steps {
        sh 'pwd && ls -l'
        script {
          def remoteConfig = [:]
          remoteConfig.name = "my-remote-server"
          remoteConfig.host = "${REMOTE_HOST}"
          remoteConfig.allowAnyHosts = true

          node {
            // 使用当前项目下的凭据管理中的 用户名 + 密码 凭据
            withCredentials([usernamePassword(
              credentialsId: "${REMOTE_CRED}",
              passwordVariable: 'password',
              usernameVariable: 'userName'
            )]) {

              // SSH 登陆用户名
              remoteConfig.user = userName
              // SSH 登陆密码
              remoteConfig.password = password

              // 请确保远端环境中有 Docker 环境
              sshCommand(
                remote: remoteConfig,
                command: "docker login -u ${CODING_DOCKER_REG_USERNAME} -p ${CODING_DOCKER_REG_PASSWORD} ${CODING_DOCKER_REG_HOST}",
                sudo: true,
              )

              // 生成devops目录
              sshCommand(
                remote: remoteConfig,
                command: "mkdir -p  /mnt/devops/${CODING_DOCKER_IMAGE_NAME}/",
                sudo: false,
              )

              sshPut(
                remote: remoteConfig,
                // 本地文件或文件夹
                from: '/root/workspace/deploy-devops.sh',
                // 远端文件或文件夹
                into: "/mnt/devops/${CODING_DOCKER_IMAGE_NAME}/"
              )

              // DOCKER_IMAGE_VERSION 中涉及到 GIT_LOCAL_BRANCH / GIT_TAG / GIT_COMMIT 的环境变量的使用
              // 需要在本地完成拼接后，再传入到远端服务器中使用
              DOCKER_IMAGE_URL = sh(
                script: "echo ${CODING_DOCKER_REG_HOST}/${CODING_DOCKER_IMAGE_NAME}:${DOCKER_IMAGE_VERSION}",
                returnStdout: true
              )

              sh(
                script:"echo ${SERVER_PREFIX_NAME}"
              )

              sshCommand(
                remote: remoteConfig,
                command: "/bin/bash /mnt/devops/${CODING_DOCKER_IMAGE_NAME}/deploy-devops.sh ${DOCKER_IMAGE_URL} ${SERVER_PREFIX_NAME} http",
                sudo: false,
              )


              echo "部署成功"
            }

          }
        }

      }
    }
  }
  environment {
    CODING_DOCKER_REG_HOST = "${CCI_CURRENT_TEAM}-docker.pkg.${CCI_CURRENT_DOMAIN}"
    CODING_DOCKER_IMAGE_NAME = "${PROJECT_NAME.toLowerCase()}/${DOCKER_REPO_NAME}/${DOCKER_IMAGE_NAME}"
    CODING_DOCKER_REG_USERNAME = '制品仓库账号'
    CODING_DOCKER_REG_PASSWORD = '制品仓库密码'
  }
}

deploy-devops.sh 脚本

#!/bin/bash

DOCKER_IMAGE_URL=${1}
SERVER_PREFIX_NAME=${2}
TASK=${3}

if [ "${SERVER_PREFIX_NAME}" = "" ]; then
    SERVER_PREFIX_NAME="server_app"
fi

if [ "${TASK}" = "" ]; then
    TASK="http"
fi

echo "DOCKER_IMAGE_URL: ${DOCKER_IMAGE_URL}"
echo "SERVER_PREFIX_NAME: ${SERVER_PREFIX_NAME}"
echo "TASK: ${TASK}"


# 检查 http服务端口
check_port() {
    # shellcheck disable=SC1083
    code=$(curl -I -m 10 -o /dev/null -s -w %{http_code}  127.0.0.1:"$1"/check_port)
    echo "$code"
}

# 检查 socket服务端口
socket_io_check_port() {
    port=$1
    # shellcheck disable=SC2196
    TCP=$(netstat -an | egrep ":${port}" | awk '$1 == "tcp" && $NF == "LISTEN" ' |wc -l)
    # shellcheck disable=SC2196
    UDP=$(netstat -an | egrep ":${port}" | awk '$1 == "udp" && $NF == "0.0.0.0:*"' |wc -l)

    ((Total = TCP + UDP ))

    if [ "${Total}" == 0 ];
      then
        echo 400
      else
        echo 200
    fi
}

#TODO HTTP服务 随机可用端口
ports=(9791 9792)
#TODO 随机API可用端口
online_port=0
reload_port=0

for i in "${!ports[@]}";
do
    code=$(check_port "${ports[$i]}")
    # shellcheck disable=SC2053
    if [[ '200' == $code ]]; then
        online_port=${ports[$i]}
    else
        reload_port=${ports[$i]}
    fi
done

if [[ $online_port == 0 ]]; then
    echo '服务未启动'
    reload_port=${ports[0]}
    echo "自动分配端口 $reload_port"
else
    echo "当前端口 $online_port"
    echo "重启端口 $reload_port"
fi

if [[ $reload_port == 0 ]]; then
	echo '部署失败, 未获取到可用端口'
	exit 1;
fi


if [[ ${TASK} == "socket" ]]; then
    #TODO 随机socket.io可用端口
    socket_io_ports=(9990 9991)
    socket_io_online_port=0
    socket_io_reload_port=0
    for i in "${!socket_io_ports[@]}";
    do
        code=$(socket_io_check_port "${socket_io_ports[$i]}")
        # shellcheck disable=SC2053
        if [[ '200' == $code ]]; then
            socket_io_online_port=${socket_io_ports[$i]}
        else
            socket_io_reload_port=${socket_io_ports[$i]}
        fi
    done
    if [[ $socket_io_online_port == 0 ]]; then
        echo '服务未启动'
        socket_io_reload_port=${socket_io_ports[0]}
        echo "socket.io自动分配端口 $socket_io_reload_port"
    else
        echo "socket.io当前端口 $socket_io_online_port"
        echo "socket.io重启端口 $socket_io_reload_port"
    fi

    if [[ $socket_io_reload_port == 0 ]]; then
      echo '部署失败, 未获取到可用socket.io端口'
      exit 1;
    fi
fi
# nginx配置文件路径
nginx_upstream='/www/server/panel/vhost/upstream/'"${SERVER_PREFIX_NAME}"'_upstream.conf'
# 创建文件
if [ ! -f "$nginx_upstream" ]; then
    touch $nginx_upstream
    upstream_default='
            upstream '"${SERVER_PREFIX_NAME}"'_upstream {
                server 127.0.0.1:'"$reload_port"';
            }
            upstream '"${SERVER_PREFIX_NAME}"'_upstream_socket_io {
                server 127.0.0.1:'"$socket_io_reload_port"';
            }
        '
    echo "$upstream_default" > $nginx_upstream
fi
# shellcheck disable=SC2016
if [[ ${TASK} == "socket" ]]; then
    upstream='
        upstream '"${SERVER_PREFIX_NAME}"'_upstream {
            server 127.0.0.1:'"$reload_port"';
        }
        upstream '"${SERVER_PREFIX_NAME}"'_upstream_socket_io {
            server 127.0.0.1:'"$socket_io_reload_port"';
        }
    '
else
    upstream='
        upstream '"${SERVER_PREFIX_NAME}"'_upstream {
            server 127.0.0.1:'"$reload_port"';
        }
    '
fi

docker pull $DOCKER_IMAGE_URL

if [[ ${TASK} == "socket" ]]; then
    echo "docker run -p $reload_port:9501 -p $socket_io_reload_port:9510 --name "${SERVER_PREFIX_NAME}"_$reload_port_$socket_io_reload_port   -dit $DOCKER_IMAGE_URL"
    docker run -p "$reload_port":9501 -p $socket_io_reload_port:9510 --name "${SERVER_PREFIX_NAME}"_"$reload_port"_"$socket_io_reload_port"   -dit $DOCKER_IMAGE_URL
    docker_container_name="${SERVER_PREFIX_NAME}"_"$reload_port"_"$socket_io_reload_port"

    old_docker_container_name="${SERVER_PREFIX_NAME}"_"$online_port"_"$socket_io_online_port"
else
    echo "docker run -p $reload_port:9501  --name "${SERVER_PREFIX_NAME}"_$reload_port   -dit $DOCKER_IMAGE_URL"
    docker run -p "$reload_port":9501  --name "${SERVER_PREFIX_NAME}"_"$reload_port"   -dit $DOCKER_IMAGE_URL
    docker_container_name="${SERVER_PREFIX_NAME}"_"$reload_port"

    old_docker_container_name="${SERVER_PREFIX_NAME}"_"$online_port"
fi
echo "新容器名称: "$docker_container_name
echo "旧容器名称: "$old_docker_container_name

while :
do
    reload=$(check_port "$reload_port")
    echo "check ${SERVER_PREFIX_NAME}_$reload_port 服务 reload $reload"

    reloadCode=0
    if [[ '200' == $reload ]]; then
        reloadCode=200
    fi

    if [[ ${TASK} == "socket" ]]; then
        reloadCode=0
        socket_io_reload=$(socket_io_check_port "$socket_io_reload_port")
        echo "check socket.io—port $socket_io_reload_port 服务 reload $socket_io_reload"
        if [[ '200' == $reload && '200' == $socket_io_reload ]]; then
            reloadCode=200
        fi
    fi
    echo "reloadCode: "$reloadCode
    echo "新容器名称: "$docker_container_name
    echo "旧容器名称: "$old_docker_container_name
    # shellcheck disable=SC2053
    if [[ '200' == $reloadCode ]]; then
        echo "新服务重启成功 ""$docker_container_name"
        echo "$upstream" > $nginx_upstream
        nginx -s reload
        if [[ $online_port != 0 ]]; then
            # 无法探知老服务持有链接响应完成时间
            sleep 5
            echo "停止旧服务 ""$old_docker_container_name"
            # shellcheck disable=SC2027
            docker stop "$old_docker_container_name"
            docker rm -f "$old_docker_container_name"
        fi
        echo "清除docker无用镜像"
        docker rmi -f $(docker images | grep "<none>" | awk "{print \$3}")
        break
    fi
    sleep 2
done
