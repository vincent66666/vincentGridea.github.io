laravel еӯҰд№ д№Ӣи·Ҝ ж•°жҚ®еә“ж“ҚдҪң ж•°жҚ®жҸ’е…ҘдёҺж•°жҚ®еЎ«е……

еүҚйқўеӯҰдәҶ Migrations иҝҒз§»е»әдәҶж•°жҚ®иЎЁпјҢзҺ°еңЁеҸҜд»ҘеӯҰд№ ж•°жҚ®еә“ж“ҚдҪңдәҶгҖӮ

ж•°жҚ®жҸ’е…Ҙ

е…ҲжҗһдёӘжҺ§еҲ¶еҷЁ

php artisan make:controller DbController

еҪ“з„¶и·Ҝз”ұд№ҹдёҚиғҪеҝҳи®°

Route::prefix('db')->group(function () {
    Route::get('insert', 'DbController@insert');
});

жҺҘдёӢжқҘз»ҷжҺ§еҲ¶еҷЁеҶҷдёӘ insert ж–№жі•

<?php

namespace App\Http\Controllers;

use Illuminate\Http\Request;
use Illuminate\Support\Facades\DB;

class DbController extends Controller
{
    //
    function insert(){
        DB::insert('INSERT INTO test (testID,title,email,`describe`) VALUES (1,"иҝҷжҳҜдёӘtitle","123456@qq.com","иҝҷжҳҜдёӘdescribe")');
    }
}

иҝҷйҮҢйңҖиҰҒеј•е…ҘдёӘ DB зұ»пјҢDB зұ»е°ұжҳҜжҲ‘д»¬ж“ҚдҪңж•°жҚ®еә“зҡ„е…ій”®пјҢDb зұ»жңүдёӘ insert ж–№жі•пјҢе®ғеҸҜд»ҘзӣҙжҺҘдј жҸ’е…ҘеҶ…е®№зҡ„ sql иҜӯеҸҘпјҢзҺ°еңЁжҲ‘и®ҝй—®дёҖ дёӢ study.laraveltest.com/db/insert еҰӮжһңжІЎжңүжҠҘй”ҷж•°жҚ®е°ұжҸ’е…ҘжҲҗеҠҹдәҶгҖӮ

зЎ®е®һжҳҜеҫҲз®ҖеҚ•еҗ§пјҒдҪҶжҳҜе№¶дёҚжҸҗеҖЎиҝҷз§ҚжӢјжҺҘеӯ—з¬ҰдёІзҡ„ж–№ејҸжқҘж·»еҠ ж•°жҚ®пјҢз»ҙжҠӨиө·жқҘйә»зғҰдёҚиҜҙиҝҳдёҚе®үе…ЁпјҢжүҖд»ҘжҲ‘жҚўз§Қж–№ејҸпјҢDB иҝҳжңүдёӘ table ж–№жі•еҸҜд»Ҙдј иЎЁеҗҚеҶҚи°ғз”Ё insert ж–№жі•еҸҜд»Ҙдј иҰҒжҸ’е…Ҙзҡ„ж•°з»„пјҢдәҺжҳҜжҲ‘ж”№йҖ йӮЈд№ҲдёҖдёўдёўдёӢйқўе°ұжҳҜж”№йҖ еҗҺзҡ„ж ·еӯҗгҖӮ

class DbController extends Controller

жҲ‘еңЁи®ҝй—®дёҖ дёӢ study.laraveltest.com/db/insert з„¶еҗҺжҹҘзңӢж•°жҚ®еә“

ж•°жҚ®еЎ«е……

дҪҝз”Ё DB жҸ’е…Ҙж•°жҚ®зҡ„ж–№ејҸеӯҰдјҡеҗҺжҲ‘д»¬е°ұеҸҜд»Ҙи®ІзӮ№еЎ«е……дәҶпјҢд»ҘеүҚдёӢиҪҪејҖжәҗйЎ№зӣ®еҰӮжһңжІЎжңүз»ҷеҮәе®үиЈ…жӯҘйӘӨпјҢе“ӘдёҖиҲ¬йғҪе…Ҳжүҫ sql ж–Үд»¶пјҢдҪҶжҳҜ sql ж–Үд»¶дёӯдёҖиҲ¬дёҚеҸӘжҳҜиЎЁз»“жһ„дёҖиҲ¬иҝҳжңүдёҖдәӣеҲқе§ӢеҢ–зҡ„ж•°жҚ®жҳЁеӨ©еӯҰд№ дәҶиЎЁз»“жһ„д»ҠеӨ©жқҘзҺ©зҺ©ж•°жҚ®пјҢеҲҡеҲҡеӯҰд№ дәҶжҸ’е…Ҙж•°жҚ®зҡ„ж–№ејҸпјҢдҪҶйӮЈжҳҜеңЁжҺ§еҲ¶еҷЁйҮҢзҡ„иҝҷз§ҚеҲӣе»әи·Ҝз”ұеҲӣе»әжҺ§еҲ¶еҷЁеҶҷеЎ«е……зҡ„ж–№ејҸдёҚеӨӘдјҳйӣ…пјҢlaravel жҳҜдјҳйӣ…зҡ„жЎҶжһ¶иҝҷд№ҲеҒҡи·ҹ laravel зҡ„дјҳйӣ…дёҚз¬ҰпјҢжүҖжңү laravel еҝ…е®ҡеҮҶеӨҮдәҶе…¶д»–ж–№жі•пјҢдёӢйқўеӯҰдёӢзҡ„еЎ«е……е°ұжҳҜеҗ‘ж•°жҚ®еә“жү№йҮҸж·»еҠ ж•°жҚ®зҡ„пјҢз®ҖеҚ•жҰӮиҝ°дёӢе°ұжҳҜ ж•°жҚ®еә“иҝҒз§»+ж•°жҚ®еЎ«е…… = SQL

жҲ‘д»¬е…ҲжқҘзңӢзңӢе®ҳзҪ‘зҡ„д»Ӣз»Қ

Laravel еҢ…еҗ«дёҖдёӘеЎ«е……зұ»еҸҜд»ҘдёәдҪ зҡ„ж•°жҚ®еә“еЎ«е……жөӢиҜ•ж•°жҚ®пјҢжүҖжңүзҡ„еЎ«е……зұ»йғҪж”ҫеңЁ database/seeds зӣ®еҪ•дёӢгҖӮдҪ еҸҜд»ҘйҡҸж„ҸдёәеЎ«е……зұ»е‘ҪеҗҚпјҢдҪҶжҳҜжӣҙе»әи®®жӮЁйҒөе®Ҳзұ»дјј UsersTableSeeder зҡ„е‘ҪеҗҚи§„иҢғгҖӮйҖҡеёёпјҢ Laravel й»ҳи®Өе®ҡд№үдәҶдёҖдёӘ DatabaseSeeder зұ»гҖӮйҖҡиҝҮиҝҷдёӘзұ»пјҢдҪ еҸҜд»Ҙз”Ё call ж–№жі•жқҘиҝҗиЎҢе…¶е®ғзҡ„ seed зұ»д»ҺиҖҢжҺ§еҲ¶ж•°жҚ®еЎ«е……зҡ„йЎәеәҸгҖӮ

зј–еҶҷ Seeders

йӮЈд№Ҳе°ұе…ҲжҗһдёӘ Seeders зұ»зҺ©зҺ© пјҢиҝҗиЎҢ Artisan е‘Ҫд»Ө make:seeder з”ҹжҲҗ Seeder.

php artisan make:seeder TestTableSeeder

жү§иЎҢдёҠйқўиҝҷдёӘе‘Ҫд»ӨеҰӮжһңжҙ—иҝҮи„ёдјҡз”ҹжҲҗдёҖдёӘ database/seeds/TestTableSeeder.php ж–Үд»¶

еҸҜд»ҘзңӢеҲ°йҮҢйқўжңүдёӘ run ж–№жі•иҝҷдёӘж–№жі•дјҡеңЁжү§иЎҢ db:seed иҝҷдёӘ Artisan е‘Ҫд»Ө ж—¶иў«и°ғз”ЁпјҢеҲ©з”Ё DB зҡ„ж–№жі•еҶҷдёҠиҰҒеЎ«е……зҡ„еҶ…е®№пјҢдҪ д№ҹеҸҜд»Ҙз”Ё жҹҘиҜўжһ„йҖ еҷЁ жҲ– Eloquent жЁЎеһӢе·ҘеҺӮ жқҘжүӢеҠЁжҸ’е…Ҙж•°жҚ®гҖӮ

TipпјҡдҪҝз”Ёж•°жҚ®еЎ«е……ж—¶дјҡиҮӘеҠЁзҰҒз”Ё жү№йҮҸиөӢеҖјдҝқжҠӨ

<?php

use Illuminate\Database\Seeder;
use Illuminate\Support\Facades\DB;

class TestTableSeeder extends Seeder
{
    /**
     * Run the database seeds.
     *
     * @return void
     */
    public function run()
    {
        $insertData = [
            [
                "testID"   => '4',
                "title"    => 'иҝҷжҳҜдёӘtitle4',
                "email"    => '4444444@qq.com',
                "describe" => 'иҝҷжҳҜдёӘdescribe4',
            ],
            [
                "testID"   => '5',
                "title"    => 'иҝҷжҳҜдёӘtitle5',
                "email"    => '55555@qq.com',
                "describe" => 'иҝҷжҳҜдёӘdescribe5',
            ],

        ];
        DB::table('test')->insert($insertData);
    }
}

иҝҗиЎҢ Seeders

еЎ«е……ж–Үд»¶е°ұз®—еҲӣе»әе®ҢжҲҗдәҶпјҢе’ҢиҝҒз§»дёҖж ·еҲӣе»әеҘҪж–Үд»¶еҶҷеҘҪеҶ…е®№е°ұеҸҜд»ҘиҝҗиЎҢдәҶпјҢжңүдёҖзӮ№иҰҒжіЁж„Ҹзҡ„жҳҜеңЁе®ҢжҲҗ seeder зұ»зҡ„зј–еҶҷд№ӢеҗҺпјҢдҪ еҸҜиғҪйңҖиҰҒдҪҝз”Ё dump-autoload е‘Ҫд»ӨйҮҚж–°з”ҹжҲҗ Composer зҡ„иҮӘеҠЁеҠ иҪҪеҷЁ

composer dump-autoload

зҺ°еңЁжҲ‘жқҘз”Ё Artisan е‘Ҫд»Ө db:seed жқҘеЎ«е……ж•°жҚ®еә“зҺ©дёҖзҺ©

php artisan  db:seed

жү§иЎҢе®Ң php artisan db:seed жҲ‘еҸ‘зҺ°ж•°жҚ®еә“йҮҢжҜӣзҡ„жІЎжңүеҗҺжқҘеҸ‘зҺ° иҝҷжҳҜеӣ дёәеЎ«е……жҜ”иҝҒз§»еӨҡдёҖдёӘжӯҘйӘӨпјҢжҲ‘еҸ‘зҺ° database/seeds зӣ®еҪ•дёӢиҝҳжңүдёӘ DatabaseSeeder.php ж–Үд»¶пјҢд»–й»ҳи®Өе°ұеӯҳеңЁдёҚжҳҜжҲ‘еҲӣе»әзҡ„жү“ејҖе®ғйҮҢйқўд№ҹжңүдёӘ run ж–№жі•е№¶дё”йҮҢйқўжңүдёҖиЎҢд»Јз ҒжіЁйҮҠжҺүдәҶ // $this->call(UsersTableSeeder::class); зҝ»зңӢдәҶ е®ҳж–№ж–ҮжЎЈ жүҚзҹҘйҒ“ db:seed е‘Ҫд»ӨдјҡеҺ»иҝҗиЎҢ DatabaseSeeder зұ»пјҢеңЁиҝҷдёӘзұ»еҸҜд»Ҙз”ЁжқҘи°ғз”Ёе…¶е®ғ Seed зұ» д№ҹе°ұжҳҜжҲ‘д»¬йңҖиҰҒд»ҝз…§жіЁйҮҠзҡ„зӨәдҫӢеңЁ DatabaseSeeder зұ»дёӯеҺ»и°ғз”ЁжҲ‘еҲҡеҲҡеҲӣе»әзҡ„Seed зұ» TestTableSeeder гҖӮйӮЈд№ҲжҲ‘еҺ»дҝ®ж”№дёҖдёӢ DatabaseSeeder зұ»

<?php

use Illuminate\Database\Seeder;

class DatabaseSeeder extends Seeder
{
    /**
     * Seed the application's database.
     *
     * @return void
     */
    public function run()
    {
        // $this->call(UsersTableSeeder::class);
         $this->call(TestTableSeeder::class);
    }
}

дёҚиҝҮиҝҷйҮҢйңҖиҰҒдё»иҰҒдёҖдёӢеЎ«е……ж–Үд»¶зҡ„жү§иЎҢйЎәеәҸпјҢжҲ‘д»¬дјҡеҲӣе»әеҫҲеӨҡеЎ«е……ж–Үд»¶еҰӮжһңжңүеӨ–й”®зҡ„иҜқжү§иЎҢйЎәеәҸдёҚеҜ№жҳҜдјҡжҠҘй”ҷзҡ„жүҖд»ҘйңҖиҰҒдёҘж јзҡ„иҮӘе·ұжҢҮе®ҡйЎәеәҸпјҢOKжҲ‘д»¬еңЁжү§иЎҢдёҖдёӢ php artisan db:seed е‘Ҫд»Ө


ж•°жҚ®з»ҲдәҺеҶҷе…Ҙж•°жҚ®еә“дёӯдәҶпјҢеҪ“з„¶ е®ҳж–№ж–ҮжЎЈ д№ҹз»ҷеҮәдәҶе…¶д»–зҡ„ж–№ејҸиҝҗиЎҢжҜ”еҰӮеҸҜд»ҘдҪҝз”Ё --class йҖүйЎ№жқҘжҢҮе®ҡдёҖдёӘзү№е®ҡзҡ„ seeder зұ»гҖӮ

php artisan db:seed --class=TestTableSeeder

д№ҹеҸҜд»Ҙз”Ё migrate:refresh иҝҷдёӘе‘Ҫд»ӨжқҘеЎ«е……ж•°жҚ®еә“пјҢиҜҘе‘Ҫд»Өдјҡеӣһж»ҡе№¶йҮҚж–°иҝҗиЎҢжүҖжңүиҝҒз§»гҖӮиҝҷдёӘе‘Ҫд»ӨеҸҜд»Ҙз”ЁжқҘйҮҚе»әж•°жҚ®еә“

php artisan migrate:refresh --seed

еҸҰеӨ–еңЁз”ҹдә§зҺҜеўғдёӯејәеҲ¶дҪҝз”ЁдёҖдәӣеЎ«е……ж“ҚдҪңеҸҜиғҪдјҡеҜјиҮҙеҺҹжңүж•°жҚ®зҡ„жӣҙж–°жҲ–дёўеӨұгҖӮдёәдәҶдҝқжҠӨз”ҹдә§зҺҜеўғж•°жҚ®еә“зҡ„ж•°жҚ®пјҢеңЁиҝҗиЎҢеЎ«е……е‘Ҫд»ӨеүҚдјҡиҝӣиЎҢзЎ®и®ӨгҖӮеҸҜд»Ҙж·»еҠ  --force йҖүйЎ№жқҘејәеҲ¶иҝҗиЎҢеЎ«е……е‘Ҫд»Ө:

php artisan db:seed --force

зӣёжҜ”дәҺиҝҒз§»пјҢеЎ«е……зҡ„д»·еҖје°ұжІЎжңүйӮЈд№Ҳйқ©е‘ҪжҖ§дәҶпјҢиҖҢдё”еЎ«е……жӣҙејәеӨ§зҡ„еҠҹиғҪйңҖиҰҒжЁЎеһӢзҡ„ж”ҜжҢҒиҝҷдёӘеҗҺйқўдјҡеӯҰеҲ°зҡ„гҖӮ
